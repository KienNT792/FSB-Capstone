1. Create `scripts/select_rtp_projects.py`:

```python
#!/usr/bin/env python3
"""
Select RTPTorrent projects based on failure rate and build count.
Usage: python scripts/select_rtp_projects.py --rtp-path data/rtp-torrent
"""
import argparse
import os
import subprocess
import pandas as pd
from pathlib import Path


def compute_project_stats(rtp_path: str) -> pd.DataFrame:
    """Scan all project CSVs and compute failure rate + build count."""
    rtp_dir = Path(rtp_path)
    results = []

    for csv_file in sorted(rtp_dir.glob("*.csv")):
        name = csv_file.stem
        # Skip non-project files
        if "-patches" in name or name.startswith("tr_"):
            continue
        # Check patches file exists (required for Sprint 2)
        patches_file = rtp_dir / f"{name}-patches.csv"
        has_patches = patches_file.exists()

        try:
            df = pd.read_csv(csv_file, low_memory=False)
            total_rows = len(df)
            if total_rows == 0:
                continue

            # Derive failure: failures > 0 OR errors > 0
            failed = ((df["failures"] > 0) | (df["errors"] > 0)).sum()
            failure_rate = failed / total_rows

            # Distinct builds = distinct job_ids
            n_builds = df["job_id"].nunique() if "job_id" in df.columns else 0

            results.append({
                "project": name,
                "total_rows": total_rows,
                "n_builds": n_builds,
                "failure_rate": round(failure_rate, 4),
                "has_patches": has_patches,
            })
        except Exception as e:
            print(f"  [WARN] Failed to read {csv_file.name}: {e}")

    return pd.DataFrame(results).sort_values("failure_rate", ascending=False)


def select_projects(df: pd.DataFrame) -> pd.DataFrame:
    """Apply selection criteria."""
    return df[
        (df["failure_rate"] >= 0.02) &
        (df["n_builds"] >= 100) &
        (df["has_patches"] == True)
    ]


def clone_repo(project: str, target_dir: Path) -> bool:
    """Clone GitHub repo for project. project format: 'user@repo'."""
    if "@" not in project:
        print(f"  [SKIP] Cannot parse project name: {project}")
        return False

    user, repo = project.split("@", 1)
    url = f"https://github.com/{user}/{repo}.git"
    dest = target_dir / project

    if dest.exists():
        print(f"  [EXISTS] {dest} — skipping clone")
        return True

    print(f"  [CLONE] {url} → {dest}")
    result = subprocess.run(
        ["git", "clone", "--depth=500", url, str(dest)],
        capture_output=True, text=True
    )
    if result.returncode != 0:
        print(f"  [ERROR] Clone failed: {result.stderr[:200]}")
        return False
    return True


def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--rtp-path", default="data/rtp-torrent",
                        help="Path to RTPTorrent CSV directory")
    parser.add_argument("--repos-dir", default="data/repos",
                        help="Directory to clone repos into")
    parser.add_argument("--no-clone", action="store_true",
                        help="Skip git clone step")
    args = parser.parse_args()

    print(f"Scanning {args.rtp_path} ...")
    stats_df = compute_project_stats(args.rtp_path)

    print("\n=== All projects (sorted by failure rate) ===")
    print(stats_df.to_string(index=False))

    selected = select_projects(stats_df)
    print(f"\n=== Selected projects (failure_rate ≥ 2%, builds ≥ 100, has patches) ===")
    print(selected.to_string(index=False))

    if len(selected) < 3:
        print(f"\n[WARN] Only {len(selected)} projects selected — minimum is 3. "
              "Consider lowering failure_rate threshold to 1%.")

    # Write summary markdown
    summary_path = Path("data/rtp-project-summary.md")
    summary_path.parent.mkdir(parents=True, exist_ok=True)
    with open(summary_path, "w") as f:
        f.write("# RTPTorrent Project Summary\n\n")
        f.write("## All Projects\n\n")
        f.write(stats_df.to_markdown(index=False))
        f.write("\n\n## Selected Projects\n\n")
        f.write(selected.to_markdown(index=False))
        f.write("\n\n## Selection Criteria\n")
        f.write("- failure_rate ≥ 0.02\n")
        f.write("- n_builds ≥ 100\n")
        f.write("- has_patches = True\n")
    print(f"\nSummary written to {summary_path}")

    # Clone repos
    if not args.no_clone:
        repos_dir = Path(args.repos_dir)
        repos_dir.mkdir(parents=True, exist_ok=True)
        print(f"\nCloning selected repos into {repos_dir} ...")
        for _, row in selected.iterrows():
            clone_repo(row["project"], repos_dir)
    else:
        print("\n[INFO] --no-clone flag set, skipping git clone.")

    print("\nDone.")


if __name__ == "__main__":
    main()
```

### S1-06 · RTPTorrent CSV loader — SQLite ingestion

**Priority:** Critical | **Estimate:** 6h

**Output files:**
- `scripts/load_rtp_dataset.py`
- `data/test_history.db` (generated, gitignored)

**Steps:**

1. Create `scripts/load_rtp_dataset.py`:

```python
#!/usr/bin/env python3
"""
Load RTPTorrent CSV files into SQLite database.

Usage:
    python scripts/load_rtp_dataset.py \
        --projects apache@sling,square@okhttp \
        --rtp-path data/rtp-torrent \
        --repos-dir data/repos \
        --db-path data/test_history.db

The script is idempotent: re-running skips already-loaded (repo, job_id, test_id) triples.
"""
import argparse
import sqlite3
import pandas as pd
import git
from pathlib import Path
from datetime import datetime


# ── Schema ──────────────────────────────────────────────────────────────────

DDL = """
CREATE TABLE IF NOT EXISTS test_runs (
    id          INTEGER PRIMARY KEY AUTOINCREMENT,
    repo        TEXT    NOT NULL,
    job_id      TEXT    NOT NULL,
    commit_sha  TEXT,
    test_id     TEXT    NOT NULL,
    outcome     TEXT    NOT NULL,     -- PASS | FAIL | ERROR | SKIPPED
    duration_ms REAL,
    timestamp   INTEGER,              -- Unix epoch from git log; NULL if unavailable
    run_index   INTEGER               -- original position in TravisCI run
);

CREATE TABLE IF NOT EXISTS file_changes (
    id          INTEGER PRIMARY KEY AUTOINCREMENT,
    repo        TEXT NOT NULL,
    commit_sha  TEXT NOT NULL,
    file_path   TEXT NOT NULL
);

CREATE UNIQUE INDEX IF NOT EXISTS ux_test_run
    ON test_runs(repo, job_id, test_id);

CREATE INDEX IF NOT EXISTS idx_commit
    ON test_runs(commit_sha);

CREATE INDEX IF NOT EXISTS idx_test
    ON test_runs(test_id);

CREATE INDEX IF NOT EXISTS idx_fc
    ON file_changes(commit_sha);
"""


# ── Helpers ──────────────────────────────────────────────────────────────────

def derive_outcome(row) -> str:
    if row.get("failures", 0) > 0 or row.get("errors", 0) > 0:
        return "FAIL"
    total = row.get("count", 1) or 1
    skipped = row.get("skipped", 0) or 0
    if skipped > 0 and skipped >= total:
        return "SKIPPED"
    return "PASS"


def load_commit_timestamps(repo_path: str) -> dict:
    """Return {sha: unix_timestamp} for all commits in cloned repo."""
    ts_map = {}
    try:
        repo = git.Repo(repo_path)
        for commit in repo.iter_commits():
            ts_map[commit.hexsha] = commit.committed_date
    except Exception as e:
        print(f"    [WARN] Could not read git timestamps from {repo_path}: {e}")
    return ts_map


def load_commit_map(rtp_path: Path) -> dict:
    """Return {tr_job_id: sha} from tr_all_built_commits.csv."""
    commits_file = rtp_path / "tr_all_built_commits.csv"
    if not commits_file.exists():
        print(f"  [WARN] {commits_file} not found — commit_sha will be NULL for all rows")
        return {}

    df = pd.read_csv(commits_file, low_memory=False)
    # Some jobs map to multiple SHAs (matrix builds) — take first
    if "tr_job_id" not in df.columns or "sha" not in df.columns:
        print(f"  [WARN] Expected columns 'tr_job_id' and 'sha' in {commits_file.name}")
        return {}

    return dict(zip(df["tr_job_id"].astype(str), df["sha"].astype(str)))


def load_project_csv(csv_path: Path, repo: str,
                     commit_map: dict, ts_map: dict) -> pd.DataFrame:
    """Parse project CSV and return test_runs DataFrame."""
    df = pd.read_csv(csv_path, low_memory=False)

    # Normalise column names to lowercase
    df.columns = [c.lower() for c in df.columns]

    # Map job_id → commit_sha
    job_col = "job_id" if "job_id" in df.columns else "travisjobid"
    df["job_id_str"] = df[job_col].astype(str)
    df["commit_sha"] = df["job_id_str"].map(commit_map)

    # Map commit_sha → timestamp
    df["timestamp"] = df["commit_sha"].map(ts_map)

    # Derive outcome
    df["outcome"] = df.apply(derive_outcome, axis=1)

    # Build result DataFrame
    test_col = "test" if "test" in df.columns else "testname"
    result = pd.DataFrame({
        "repo":        repo,
        "job_id":      df["job_id_str"],
        "commit_sha":  df["commit_sha"],
        "test_id":     df[test_col].astype(str),
        "outcome":     df["outcome"],
        "duration_ms": df.get("duration", pd.Series(dtype=float)) * 1000
                       if "duration" in df.columns else None,
        "timestamp":   df["timestamp"],
        "run_index":   df.get("index", pd.Series(dtype=float)),
    })
    return result


def load_patches_csv(patches_path: Path, repo: str) -> pd.DataFrame:
    """Parse patches CSV and return file_changes DataFrame."""
    df = pd.read_csv(patches_path, low_memory=False)
    df.columns = [c.lower() for c in df.columns]

    sha_col = "sha" if "sha" in df.columns else "commit_id"
    file_col = "name" if "name" in df.columns else "file"

    return pd.DataFrame({
        "repo":       repo,
        "commit_sha": df[sha_col].astype(str),
        "file_path":  df[file_col].astype(str),
    })


def insert_test_runs(conn: sqlite3.Connection, df: pd.DataFrame, repo: str):
    """Insert test_runs rows, skipping duplicates."""
    rows = df[["repo", "job_id", "commit_sha", "test_id",
               "outcome", "duration_ms", "timestamp", "run_index"]].values.tolist()

    conn.executemany("""
        INSERT OR IGNORE INTO test_runs
            (repo, job_id, commit_sha, test_id, outcome, duration_ms, timestamp, run_index)
        VALUES (?, ?, ?, ?, ?, ?, ?, ?)
    """, rows)
    conn.commit()


def insert_file_changes(conn: sqlite3.Connection, df: pd.DataFrame):
    """Insert file_changes rows."""
    rows = df[["repo", "commit_sha", "file_path"]].values.tolist()
    conn.executemany("""
        INSERT OR IGNORE INTO file_changes (repo, commit_sha, file_path)
        VALUES (?, ?, ?)
    """, rows)
    conn.commit()


# ── Main ─────────────────────────────────────────────────────────────────────

def main():
    parser = argparse.ArgumentParser(description="Load RTPTorrent into SQLite")
    parser.add_argument("--projects", required=True,
                        help="Comma-separated list of project names, e.g. apache@sling,square@okhttp")
    parser.add_argument("--rtp-path", default="data/rtp-torrent",
                        help="Path to RTPTorrent CSV directory")
    parser.add_argument("--repos-dir", default="data/repos",
                        help="Path to directory containing cloned repos")
    parser.add_argument("--db-path", default="data/test_history.db",
                        help="Output SQLite database path")
    args = parser.parse_args()

    projects = [p.strip() for p in args.projects.split(",")]
    rtp_path = Path(args.rtp_path)
    repos_dir = Path(args.repos_dir)
    db_path = Path(args.db_path)
    db_path.parent.mkdir(parents=True, exist_ok=True)

    # Connect and initialise schema
    conn = sqlite3.connect(str(db_path))
    conn.executescript(DDL)
    print(f"Database: {db_path}")

    # Load commit map once (shared across projects)
    print("\nLoading commit SHA map from tr_all_built_commits.csv ...")
    commit_map = load_commit_map(rtp_path)
    print(f"  {len(commit_map):,} job→sha mappings loaded")

    for project in projects:
        print(f"\n{'='*60}")
        print(f"Project: {project}")

        csv_path = rtp_path / f"{project}.csv"
        patches_path = rtp_path / f"{project}-patches.csv"
        repo_path = repos_dir / project

        if not csv_path.exists():
            print(f"  [ERROR] CSV not found: {csv_path}")
            continue

        # Load git timestamps for this project
        ts_map = {}
        if repo_path.exists():
            print(f"  Loading git timestamps from {repo_path} ...")
            ts_map = load_commit_timestamps(str(repo_path))
            print(f"  {len(ts_map):,} commit timestamps resolved")
        else:
            print(f"  [WARN] Repo not cloned at {repo_path} — timestamp will be NULL")

        # Load test runs
        print(f"  Loading {csv_path.name} ...")
        runs_df = load_project_csv(csv_path, project, commit_map, ts_map)
        print(f"  {len(runs_df):,} rows parsed")

        null_sha_rate = runs_df["commit_sha"].isna().mean()
        failure_rate  = (runs_df["outcome"] == "FAIL").mean()
        print(f"  Null SHA rate:  {null_sha_rate:.1%}")
        print(f"  Failure rate:   {failure_rate:.1%}")

        if null_sha_rate > 0.40:
            print(f"  [WARN] Null SHA rate > 40% — commit-level features will be limited for this project")

        insert_test_runs(conn, runs_df, project)
        print(f"  test_runs inserted (duplicates skipped)")

        # Load file changes
        if patches_path.exists():
            print(f"  Loading {patches_path.name} ...")
            fc_df = load_patches_csv(patches_path, project)
            insert_file_changes(conn, fc_df)
            print(f"  {len(fc_df):,} file_changes rows inserted")
        else:
            print(f"  [WARN] No patches file found: {patches_path.name}")

    # Summary
    print(f"\n{'='*60}")
    print("Summary:")
    total_runs  = conn.execute("SELECT COUNT(*) FROM test_runs").fetchone()[0]
    total_fails = conn.execute("SELECT COUNT(*) FROM test_runs WHERE outcome='FAIL'").fetchone()[0]
    total_fc    = conn.execute("SELECT COUNT(*) FROM file_changes").fetchone()[0]
    print(f"  test_runs rows:    {total_runs:,}")
    print(f"  FAIL rows:         {total_fails:,}  ({total_fails/max(total_runs,1):.1%})")
    print(f"  file_changes rows: {total_fc:,}")

    conn.close()
    print(f"\nDone. Database saved to {db_path}")


if __name__ == "__main__":
    main()
```
